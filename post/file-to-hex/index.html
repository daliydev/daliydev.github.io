import java.io.File;
import java.io.FileInputStream;
import java.io.IOException;
import java.io.InputStream;

public class Main {
    public static void main(String[] args) throws IOException {
        File file = new File("/.jpg");
        if(file.exists()){
            System.out.println(convertFileToHex(file));
        }
    }
    
    public static String convertFileToHex(File file) throws IOException {
        StringBuilder result = new StringBuilder();
        StringBuilder hex = new StringBuilder();
        StringBuilder input = new StringBuilder();
    
        int count = 0;
        int value;
        
        try (InputStream inputStream = new FileInputStream(file)) {
        
            while ((value = inputStream.read()) != -1) {
                // 16진수로 변환
                hex.append(String.format("%02X ", value));
                
                // 문자를 변경할 수 없는 경우
                if (!Character.isISOControl(value)) {
                    input.append((char) value);
                } else {
                    input.append(".");
                }
                
                if (count == 14) {
                    result.append(String.format("%-60s | %s%n", hex, input));
                    hex.setLength(0);
                    input.setLength(0);
                    count = 0;
                    break; // 첫 라인만 출력
                } else {
                    count++;
                }
            }
            
            if (count > 0) {
                result.append(String.format("%-60s | %s%n", hex, input));
            }
        }
        
        return result.toString();
    }
}

    File file = new File(System.currentTimeMillis() + "." + url.getFile().substring(url.getFile().lastIndexOf(".")));

    URLConnection urlConnection = url.openConnection();
    urlConnection.setConnectTimeout(30000);
    urlConnection.setReadTimeout(30000);
    InputStream in = urlConnection.getInputStream();
    OutputStream out = new FileOutputStream(file);
    
    int readLine;
    byte[] buffer = new byte[1024];
    while((readLine = in.read(buffer)) != -1){
        out.write(buffer, 0, readLine);
    }
    
    in.close();
    out.close();
}catch (IOException e) {
    e.printStackTrace();
