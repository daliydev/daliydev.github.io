import java.io.File;
import java.io.InputStream;
import java.net.URL;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.Paths;
import java.time.Duration;
import java.util.Iterator;
import java.util.List;
import org.openqa.selenium.By;
import org.openqa.selenium.Cookie;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.chrome.ChromeOptions;
import org.openqa.selenium.support.ui.ExpectedConditions;
import org.openqa.selenium.support.ui.WebDriverWait;

public class Main {
    public static String WEB_DRIVER_ID = "webdriver.chrome.driver";
    public static String WEB_DRIVER_PATH = "driver/chromedriver_103";
    public static String INSTAGRAM_SESSION_KEY = "sessionid";
    public static String INSTAGRAM_SESSION_VALUE = "";
    
    public static void main(String[] args){
        String target = args.length > 0 ? args[0] : "https://www.instagram.com/p/#";
        // 가져올 사진

        System.setProperty(WEB_DRIVER_ID, WEB_DRIVER_PATH);
        ChromeOptions options = new ChromeOptions();
        // options.addArguments("--headless", "--disable-gpu", "--no-sandbox", "--start-maximized");
        // --headless 브라우저가 뜨지 않고 실행
        // --start-maximized 브라우저 최대실행
        // GUI 제공하지 않는 환경(리눅스 서버) --disable-gpu, --no-sandbox 옵션 추가
        
        WebDriver driver = new ChromeDriver(options);
        try {
            WebDriverWait webDriverWait = new WebDriverWait(driver, Duration.ofSeconds(20));
            driver.get("https://www.instagram.com");
            driver.manage().addCookie(new Cookie(INSTAGRAM_SESSION_KEY, INSTAGRAM_SESSION_VALUE));
            driver.get(target);
            
            // 로그인 페이지로 이동 되면..
            if(driver.getCurrentUrl().indexOf("/login") > -1){
                WebElement element;
                element = driver.findElement(By.name("username"));
                element.sendKeys("{ID}");
                element = driver.findElement(By.name("password"));
                element.sendKeys("{PW}");
                element = driver.findElement(By.id("loginForm"));
                element.submit();
                
                webDriverWait.until(ExpectedConditions.visibilityOfElementLocated(By.className("kqZqN")));
                
                Iterator<Cookie> it = driver.manage().getCookies().iterator();
                while(it.hasNext()){
                    Cookie cookie = it.next();
                    if(cookie.getName().equals(INSTAGRAM_SESSION_KEY)){
                        INSTAGRAM_SESSION_VALUE =  cookie.getValue();
                        driver.manage().addCookie(new Cookie(INSTAGRAM_SESSION_KEY, INSTAGRAM_SESSION_VALUE));
                        break;
                    }
                }
                
                driver.get(target);
            }
            webDriverWait.until(ExpectedConditions.visibilityOfElementLocated(By.className("_acbh")));
            
            File file = new File("resources");
            if(!file.exists()){
                file.mkdir();
            }
            
            // 사진 복사
            List<WebElement> elementList = driver.findElements(By.tagName("img"));
            for(WebElement we : elementList){
                String image = we.getDomAttribute("src");
                String iu = image.substring(0, image.indexOf("?") > -1 ? image.indexOf("?") : image.length());
                String ext = iu.substring(iu.lastIndexOf("."));
                try(InputStream in = new URL(image).openStream()){
                    Path imagePath = Paths.get(String.format("%s/%d%s", file.getPath(), System.currentTimeMillis(), ext));
                    Files.copy(in, imagePath);
                }
            }
        }catch (Exception e) {
            e.printStackTrace();
        } finally {
             driver.quit();
        }
    }
}

WebDriverWait.until 해당 element가 로딩될 때까지 대기할 수 있도록.

ExpectedConditions.visibilityOf(WebElement) dom에 element 존재 여부 (요소의 존재를 확인하지 않습니다) ExpectedConditions.visibilityOfElementLocated(By) dom에 element 존재 여부, 화면 상에 보여지는지까지 체크. ExpectedConditions.presenceOfElementLocated(By) dom에 존재하는지만 체크.
