[java] Adapter 패턴

0001-01-01
java

서로 다른 인터페이스가 호환이 되지 않을 때
인터페이스에 영향 받는 부분들을 수정하다가 시스템 장애가 발생하는 일이 발생할 수 있기 때문에,
어댑터 클래스를 만들어 기존 로직에 영향 없이 호환되게 한다.

예시 1)

예시) 코틀린 함수를 Java 언어의 함수로 대체 하려고 한다.

public class Adapter extends KotlinLanguage {
    private JavaLanguage javaLanguage;
    
    public Adapter(JavaLanguage javaLanguage) {
        this.javaLanguage = javaLanguage;
    }
    
    @Override
    public int sum(int a, int b) {
        return javaLanguage.sum(a, b);
    }
}

    }

package adapter;

public class LangAdapter implements LangPackage {
    private Language language;
    
    public LangAdapter(Language language) {
        this.language = language;
    }
    
    @Override
    public void install() {
        language.execute();
    }
}

public class Java implements Language {
    
    @Override
    public void execute() {
        System.out.println("java");
    }
}

public class Kotlin implements Language {
    
    @Override
    public void execute() {
        System.out.println("kotlin");
    }
}

public class Swift implements Language {
    
    @Override
    public void execute() {
        System.out.println("swift");
    }
}
