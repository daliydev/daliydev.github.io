프로토타입은 생성 패턴 중 하나이다.
객체 생성하는 데 시간과 비용이 많이 들고, 이미 유사한 객체가 존재하는 경우에 사용합니다.

new를 이용해서 객체를 생성하지 말고, 오직 clone() 메서드를 이용해서만 생성한다.

package prototype;

import java.util.ArrayList;
import java.util.List;

public class Programming implements Cloneable {
    private List<String> mList;
    
    public Programming() {
        this.mList = new ArrayList<>();
        mList.add("C");
        mList.add("Java");
    }
    
    public Programming(List<String> list) {
        this.mList = list;
    }
    
    public List<String> getmList() {
        return mList;
    }
    
    @Override
    protected Object clone() throws CloneNotSupportedException {
        List<String> temp = new ArrayList<>();
        for(String lang : mList) {
            temp.add(lang);
        }
        return new Programming(temp);
    }
}

    public static void main(String[] args) throws CloneNotSupportedException {
        Programming programming = new Programming();
        Programming copy1 = (Programming) programming.clone();
        Programming copy2 = (Programming) programming.clone();
        
        copy1.getmList().add("Kotlin");
        copy2.getmList().remove("Java");
        
        System.out.println(programming.getmList());
        System.out.println(copy1.getmList());
        System.out.println(copy2.getmList());
//        [C, Java]
//        [C, Java, Kotlin]
//        [C]
    }
}

clone()을 재정의하기 위해 Cloneable 인터페이스를 구현하고, mList에 대하여 깊은 복사를 합니다.

객체를 복사하는 것이 new, 네트워크나 DB 접근보다 훨씬 비용이 적음
